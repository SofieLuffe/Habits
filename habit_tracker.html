    <script>

        const habitTable = document.getElementById('habit-table').querySelector('tbody');

        const wishTable = document.getElementById('wish-table').querySelector('tbody');


        function addHabit() {

            const name = document.getElementById('habit-name').value;

            const type = document.getElementById('habit-type').value;

            const points = parseInt(document.getElementById('habit-points').value);


            if (name && !isNaN(points)) {

                const row = habitTable.insertRow();

                row.innerHTML = `

                    <td>${name}</td>

                    <td>${type}</td>

                    <td>${points}</td>

                    <td><button onclick="deleteRow(this)">Slet</button></td>

                `;


                document.getElementById('habit-name').value = '';

                document.getElementById('habit-type').value = 'repeating';

                document.getElementById('habit-points').value = '';

            } else {

                alert('Udfyld alle felter korrekt!');

            }

        }


        function addWish() {

            const name = document.getElementById('wish-name').value;

            const cost = parseInt(document.getElementById('wish-cost').value);


            if (name && !isNaN(cost)) {

                const row = wishTable.insertRow();

                row.innerHTML = `

                    <td>${name}</td>

                    <td>${cost}</td>

                    <td><button onclick="deleteRow(this)">Slet</button></td>

                `;


                document.getElementById('wish-name').value = '';

                document.getElementById('wish-cost').value = '';

            } else {

                alert('Udfyld alle felter korrekt!');

            }

        }


        function deleteRow(button) {

            const row = button.parentElement.parentElement;

            row.parentElement.removeChild(row);

        }
